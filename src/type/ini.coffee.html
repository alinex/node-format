include base modules

 
ini = require 'ini'

object -> string 

exports.stringify = (obj, options, cb) ->
  try
    text = ini.encode obj,
      whitespace: if options?.whitespace? then options.whitespace else true
  catch error
    return cb error
  cb null, text

string -> object 

exports.parse = (text, _, cb) ->
  try
    result = ini.decode text
  catch error
    return cb error
