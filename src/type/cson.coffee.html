CSON = require 'cson-parser'

object -> string 

exports.format = (obj, options, cb) ->
  cb null, CSON.stringify obj, null, options?.indent ? 2

string -> object 

exports.parse = (text, cb) ->
  try
    result = CSON.parse text
  catch error
    return cb error
  cb null, result
