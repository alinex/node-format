string -> object 

exports.parse = (text, _, cb) ->
  try
    obj = yaml.safeLoad text
  catch error
    return cb new Error(
      error.message
      .replace 'JS-YAML: ', ''
      .replace /[\s^]+/g, ' '
    )
  cb null, obj
