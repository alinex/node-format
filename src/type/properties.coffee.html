exports.parse = (text, cb) ->
  properties.parse text,
    sections: true

  , (err, result) ->
    return cb err if err

propertiesCheck = (data) ->
  return true unless typeof data is 'object'
  for k, v of data
    if v is null or ~'[]{}/'.indexOf k.charAt 0
      return false
    return false unless propertiesCheck v
  return true
